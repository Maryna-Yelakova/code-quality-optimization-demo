Frontend Code Quality and Optimization

Teaching examples for HTML, CSS, and JavaScript. Each section compares a BAD implementation with a GOOD fixed variant that improves maintainability, accessibility, and performance.

HTML

Bad 
<div style="font-size:24px;">Product</div>
<img src="/assets/phone.jpg">
<input type="text" placeholder="name">
<button>Click</button>
Good (Fixed) 
<article>
  <h3>Phone X</h3>
  <img src="/assets/phone.jpg" alt="Phone X in black color">
  <label for="customer-name">Your name</label>
  <input id="customer-name" name="customerName" autocomplete="name">
  <button type="submit">Add to cart</button>
</article>

CSS

Bad 
#page .box1 { color:#fff; background:blue; padding:13px; }
#page .box2 { color:#fff; background:blue; padding:13px; }
#page .box3 { color:#fff !important; background:blue; padding:13px; }
Good (Fixed) 
:root {
  --color-text-inverse:#fff;
  --color-primary:#1d4ed8;
  --space-md:0.75rem;
}
.card {
  color:var(--color-text-inverse);
  background:var(--color-primary);
  padding:var(--space-md);
}

JavaScript

Bad 
// 1) Naming + var usage
function doIt(a) {
  for (var i = 0; i < a.length; i++) console.log(a[i].n);
}

// 2) Repeated DOM queries in loops
for (let i = 0; i < items.length; i++) {
  document.getElementById('result').innerHTML += '<li>' + items[i].name + '</li>';
}

// 3) Unsafe injection
document.getElementById('welcome').innerHTML = '<p>Welcome ' + userInput + '</p>';

// 4) Unnecessary polling
setInterval(() => console.log('Checking status...'), 1000);

// 5) Missing async error handling
const response = await fetch('/api/users');
const users = await response.json();

// 6) Mutating function arguments
prices.sort((a, b) => a - b);

// 7) Duplicated business logic in multiple places
if (item.vip) total += item.price * 0.8;
else total += item.price * 0.9;
Good (Fixed) 
// 1) Naming + block scoping
function logUserNames(users) {
  for (let index = 0; index < users.length; index++) console.log(users[index].name);
}

// 2) One lookup + fragment rendering
const list = document.getElementById('result');
const fragment = document.createDocumentFragment();
items.forEach((item) => {
  const li = document.createElement('li'); li.textContent = item.name; fragment.appendChild(li);
});
list.appendChild(fragment);

// 3) Safe text rendering
paragraph.textContent = `Welcome ${userInput}`;

// 4) Controlled interval + cleanup
const intervalId = setInterval(checkStatus, 5000);
setTimeout(() => clearInterval(intervalId), 30000);

// 5) Async with try/catch + response.ok
if (!response.ok) throw new Error(`Request failed: ${response.status}`);

// 6) Immutable sorting
const sorted = [...prices].sort((a, b) => a - b);

// 7) Reusable helper
const discounted = item.vip ? item.price * 0.8 : item.price * 0.9;

JavaScript Optimization

Bad 
// 1) DOM updates inside loop
for (let i = 0; i < products.length; i++) {
  list.innerHTML += "<li>" + products[i].title + "</li>";
}

// 2) Expensive filter on every keypress
items.filter((item) => item.name.toLowerCase().includes(query.toLowerCase()));

// 3) Duplicate API call (same id)
await fetch("/api/users/" + userId);

// 4) Listener per row button
document.querySelectorAll(".row button").forEach((button) => { ... });

// 5) Recalculate same cart total repeatedly
items.reduce((sum, item) => sum + item.price * item.quantity, 0);

// 6) Read/write layout in same loop
const w = card.offsetWidth; card.style.height = String(w * 0.6) + "px";

// 7) Long blocking loop
for (let i = 0; i < items.length; i++) out.push(items[i] * 2);
Good (Fixed) 
// 1) Build once, update once
const html = products.map((p) => "<li>" + p.title + "</li>").join("");
list.innerHTML = html;

// 2) Debounce expensive operation
const searchGood = debounce((items, query, onResult) => { ... }, 250);

// 3) Cache requests by id
if (userRequestCache.has(userId)) return userRequestCache.get(userId);

// 4) Event delegation
container.addEventListener("click", (event) => { ... });

// 5) Memoize totals with stable key
if (cartTotalCache.has(cacheKey)) return cartTotalCache.get(cacheKey);

// 6) Batch reads then writes
const widths = cards.map((card) => card.offsetWidth);
cards.forEach((card, i) => { card.style.height = String(widths[i] * 0.6) + "px"; });

// 7) Chunk heavy work
setTimeout(runChunk, 0);

JavaScript Arrays

Bad 
// 1) Mutates original array
scores.sort((a, b) => a - b);

// 2) map() used for side effects
users.map((u) => console.log(u.name));

// 3) Nested loops (O(n^2))
for (let i = 0; i < users.length; i++) {
  for (let j = 0; j < profiles.length; j++) { ... }
}

// 4) reduce without initial value
items.reduce((sum, item) => sum + item.price);

// 5) Multiple passes
users.filter(...).map(...).reduce(...);

// 6) Loose equality
ids.some((id) => id == targetId);
Good (Fixed) 
// 1) Immutable sort
[...scores].sort((a, b) => a - b);

// 2) forEach for side effects
users.forEach((u) => console.log(u.name));

// 3) Map index for O(1) lookup
const profileByUserId = new Map(profiles.map((p) => [p.userId, p]));

// 4) Safe reduce initialization
items.reduce((sum, item) => sum + item.price, 0);

// 5) Single-pass reduce
users.reduce((sum, user) => user.vip ? sum + user.points : sum, 0);

// 6) Strict equality
ids.some((id) => id === targetId);

JavaScript Objects

Bad 
// 1) Mutates shared object
user.name = newName;

// 2) Unsafe deep access
return user.profile.address.city;

// 3) for...in without own check
for (const key in source) { result[key] = source[key]; }

// 4) Plain object used as hash map
counts[word] = (counts[word] || 0) + 1;

// 5) Blind merge
{ ...target, ...patch }

// 6) Loose key check pattern
Object.keys(obj).includes(field);
Good (Fixed) 
// 1) Immutable object update
{ ...user, name: newName }

// 2) Optional chaining + fallback
user.profile?.address?.city ?? "Unknown city"

// 3) Own enumerable entries only
Object.fromEntries(Object.entries(source))

// 4) Use Map for dynamic keys
counts.set(word, (counts.get(word) ?? 0) + 1);

// 5) Safe merge (ignore undefined)
Object.fromEntries(Object.entries(patch).filter(([, v]) => v !== undefined))

// 6) Fast own-property check
Object.hasOwn(obj, field);
